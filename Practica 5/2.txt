procedure Cola is

    task type Cliente;

    arrClientes:array[1..N] of Cliente;

    task Empleado is 
        loop
            entry Atender(datos: in text, res: out text);
        end loop;
    end Empleado; 

    task body Cliente is
        datos,res:text; //datos inicializada
    begin
        select
            Empleado.Atender(datos,res)
        or delay(10 min)
            null
        end select
    end Cliente

    task body Empleado is
    datos,res:text;
    begin
        Accept Atender(datos,res) do
            res=Resultado(res);
        end Atender;
    end Empleado
begin

end Cola;
